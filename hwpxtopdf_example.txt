import os
import winreg
import win32com.client
import pythoncom

def register_hwp_security_module(dll_path: str):
    """레지스트리에 DLL 경로를 자동 등록 (이미지의 작업을 파이썬이 대신 함)"""
    key_path = r"Software\HNC\HwpAutomation\Modules"
    try:
        winreg.CreateKey(winreg.HKEY_CURRENT_USER, key_path)
        key = winreg.OpenKey(winreg.HKEY_CURRENT_USER, key_path, 0, winreg.KEY_WRITE)
        
        # ★ 이미지에 나온 이름 그대로 레지스트리에 등록! ★
        winreg.SetValueEx(key, "FilePathCheckerModuleExample", 0, winreg.REG_SZ, dll_path)
        winreg.CloseKey(key)
    except Exception as e:
        print(f"레지스트리 등록 실패: {e}")

def convert_hwpx_to_pdf(input_path: str, output_path: str) -> bool:
    # 1. 다운로드 받은 dll 파일의 실제 경로를 지정해주세요.
    dll_path = os.path.abspath("./FilePathCheckerModuleExample.dll") 
    
    # 2. 여기서 레지스트리 자동 등록을 실행합니다.
    register_hwp_security_module(dll_path) 

    pythoncom.CoInitialize()
    hwp = None
    try:
        hwp = win32com.client.gencache.EnsureDispatch('HWPFrame.HwpObject')
        
        # 3. ★ 한컴 공식 문서(첫 번째 이미지)에 나온 그대로 보안 모듈 적용! ★
        hwp.RegisterModule("FilePathCheckDLL", "FilePathCheckerModuleExample")
        
        hwp.XHwpWindows.Item(0).Visible = False
        
        input_abs = os.path.abspath(input_path)
        output_abs = os.path.abspath(output_path)
        
        if not hwp.Open(input_abs):
            return False
            
        res = hwp.SaveAs(output_abs, "PDF")
        return res
        
    except Exception as e:
        print(f"Error: {e}")
        return False
    finally:
        if hwp:
            try: hwp.Quit()
            except: pass
        pythoncom.CoUninitialize()
